function* sequence(){
  let count = 0;
  while(true) {
    count += 1;
    yield count;
  }
}

const generator = sequence();
generator.next().value; //1
generator.next().value; //2
generator.next().value; //3

function sequence(){
  let count = 0;
  return function(){
    count += 1;
    return count;
  }
}

const counter = (function(){
    let state = 0;
    
    function increment(){
       state += 1;
       return state;
    }
    
    function decrement(){
       state -= 1;
       return state;
    }
    
    return {
       increment,
       decrement
    }
})();
